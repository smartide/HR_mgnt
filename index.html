            var FCImport = (function($){
                  _elementId = "importUI";    //No I18N
                  _dialogId = "importdialog"; //No I18N
                  _dialogIsOpened = false;
                  return{
                      options : {
                          errorMessages: {},
                          maxSize: 1000000,
                          fileFormats:'.doc,.docx,.odt,.txt', //No I18N
                          uploadServerStatus: true,
                          uploadURL: FComponents.option('contextPath') +'import.do',  //No I18N
                          uploadServerURL:  window.location.protocol + '//uploadserver.zoho.com/postservlet',    //No I18N
                          fileOpenURL : "ropen.do?rid={0}"  //No I18N            
                      },
                      init: function(options){
                          FCImport.options = $.extend({}, FCImport.options, options);
                          FCImport.options.uploadURL = FComponents.option('contextPath') +'import.do';    //No I18N
                      },
                      open : function(options){
                          if(!_dialogIsOpened){
                              var importDialog = $('#'+_dialogId);    //No I18N
                              if(importDialog.length){
                                  importDialog.zdialog('close');  //No I18N
                              }
                              var importUI = $('#'+ _elementId);  //No I18N

                              FCImport.options = $.extend({}, FCImport.options, options);

                              importDialog = $('
').appendTo($('body'));   //No I18N
                              importUI = $('
').appendTo(importDialog);   //No I18N
                              importDialog.zdialog({modal: true,minimizable: true});
                              importUI.attr( 'ctype' , "zimport");    //No I18N
                              var importOptions = {
                                  errorMessages: FCImport.options.errorMessages,
                                  supportedFileFormats: FCImport.options.fileFormats,
                                  uploadServerStatus: FCImport.options.uploadServerStatus,
                                  uploadServerURL: FCImport.options.uploadServerURL,
                                  serviceType: FComponents.option('service'), //No I18N
                                  allowedFileSize: parseInt(FCImport.options.maxSize),
                                  passwordprotected: false,
                                  uploadURL: FComponents.option('contextPath') +'import.do',  //No I18N
                                  importLocalFileName: 'xfile',   //No I18N
                                  password: 'password',   //No I18N
                                  importURLFileName: 'importurl'  //No I18N
                              };
                              importUI.zimport(importOptions);
                              importDialog.on('dialogopen', function(ev, ui){   //No I18N
                                  _dialogIsOpened = true;
                              });
                              importDialog.zdialog('open');   //No I18N
                              importDialog.on('dialogclose', function(ev, ui){   //No I18N
                                  _dialogIsOpened = false
                                  importDialog.remove();
                              });
                          }
                      },
                      close : function(){
                          var importDialog = $('#'+ _dialogId);   //No I18N
                          if(importDialog.length){
                              importDialog.zdialog('close');  //No I18N
                          }
                      },
                      updateImportStatus : function(importResponse){
                          if(FCImport.options.uploadServerStatus){
                              $('#'+ _elementId).zimport('showMessage','status',importResponse); //No I18N
                          }
                      },
                      setErrorMessage: function(errorMessages){
                          $('#'+ _elementId).zimport('setErrorMessage',errorMessages);    //No I18N
                      },
                      setImportResourceId : function(importResponse){
                          if(FCImport.options.uploadServerStatus){
                              $('#'+ _elementId).zimport('setImportResourceId', importResponse); //No I18N
                          }
                      },
                      upload: function(file, fileObj, callbacks){
                          var defaultInfo = {
                              file: undefined,
                              newName: "",    //No I18N
                              password: ""    //No I18N
                          }
                          var uploadObj = $.extend({}, defaultInfo, {file: file, newName: fileObj.name });

                          var handlecallbacks = {
                              progress: function(serverResponse, per){
                                  if(callbacks && callbacks.progress){
                                      var responseObj = serverResponse;
                                      if(typeof responseObj !== "object"){    //No I18N
                                          responseObj = $.parseJSON(serverResponse);
                                      }
                                      callbacks.progress(responseObj, per);
                                  }
                              },
                              success: function(serverResponse){
                                  var responseObj = serverResponse;
                                  if(typeof responseObj !== "object"){    //No I18N
                                      responseObj = $.parseJSON(serverResponse);
                                  }
                                  if(callbacks){
                                      if(responseObj.status === 'success' && callbacks.success){  //No I18N
                                          callbacks.success(responseObj);
                                      }
                                      else if(callbacks.fail){
                                          callbacks.fail(responseObj);
                                      }
                                  }
                              },
                              fail: function(serverResponse) {
                                  if(callbacks && callbacks.fail){
                                      var responseObj = serverResponse;
                                      if(typeof responseObj !== "object"){    //No I18N
                                          responseObj = $.parseJSON(serverResponse);
                                      }
                                      callbacks.fail(responseObj);
                                  }
                              }
                          };
                          var data = {
                              docname: uploadObj.newName,    // need a extension with filename for serverside extension find
                              password: uploadObj.password ? uploadObj.password : '',    //No I18N
                              servicename: FComponents.option('service').toLowerCase()   //No I18N
                          };
                          if (window.FileReader && typeof file !== "string") {
                              var checkSize = (file.get(0).files[0].size > this.options.allowedFileSize)? true: false ;
                              if(!checkSize){
                                  uploadObj.documentType = file.get(0).files[0].type;
                              }
                          }
                          data[ FComponents.option('csrfParamName') ] = FCUtil.getCSRFToken();    //No I18N
                          var importFromLocal = true;
                          if(typeof file !== "string" && file.get(0) && file.get(0).files[0]){    //No I18N
                              data.xfile = file;
                              uploadObj.xfile = file;
                              //uploadObj.password = fileObj.password;
                              data.filetype = 'localfile';    //No I18N
                          }
                          else{
                              importFromLocal = false;
                              data.importurl = file;
                              data.filetype = 'weburl';   //No I18N
                          }
                          if(fileObj.password !== '' && fileObj.password !== undefined){  //No I18N
                              data.password = fileObj.password;
                          }
                          uploadObj = $.extend({}, uploadObj, this._checkFileExtn(uploadObj.newName, importFromLocal));            
                          if(FCImport.options.uploadServerStatus && importFromLocal && this._checkAjaxUploadAndProgressEvents()){
                              this._udServerUpload(uploadObj, handlecallbacks);
                          }
                          else{
                              $.ajaxUpload( FCImport.options.uploadURL, data, handlecallbacks);
                          }
                      },
                      _checkAjaxUploadAndProgressEvents: function(){
                          var xhr = new XMLHttpRequest();
                          return !! (xhr && ('upload' in xhr) && ('onprogress' in xhr.upload));   //No I18N
                      },
                      _checkFileExtn: function(fileName, importFromLocal){
                          if(importFromLocal){
                              fileName = fileName.substr(fileName.lastIndexOf('\\')+1);   //No I18N
                          }
                          var index = fileName.lastIndexOf('.');  //No I18N
                          var index = (index > 0) ? index : fileName.length;
                          var ext = fileName.substr(index);
                          return {
                              newName: fileName,
                              fileExtn: (ext ? ext.toLowerCase() : undefined)
                          };
                      },
                      _udServerUpload: function(filedetails, callbacks){

                          var xhr = new XMLHttpRequest();
                          var upload = xhr.upload;
                          upload.fileIndex = 0;
                          upload.fileName = encodeURIComponent(filedetails.newName);
                          upload.downloadStartTime = new Date().getTime();
                          upload.currentStart = upload.downloadStartTime;
                          upload.currentProgress = 0;
                          upload.startData = 0;

                          this._uploadId = upload.fileIndex + "_new_" + $.now() + "_"; // NO I18N
                          xhr.onreadystatechange = function(){
                              if (xhr.readyState === 4 &&  xhr.status !== 200){
                                  callbacks.fail(xhr);
                              }
                              else if (xhr.readyState === 1){
                                  callbacks.progress(xhr);
                              }
                          };
                          xhr.open("POST", FCImport.options.uploadServerURL, true); //No I18N
                          xhr.setRequestHeader("Connection", "Keep-Alive");   //No I18N
                          xhr.setRequestHeader("Content-type","application/x-www-form-urlencoded");   //No I18N
                          xhr.setRequestHeader("x-service", "EX");    //No I18N
                          xhr.setRequestHeader("x-streammode","1");   //No I18N
                          xhr.setRequestHeader("charset","utf-8");    //No I18N
                          xhr.setRequestHeader("upload-id", this._uploadId);  //No I18N
                          xhr.setRequestHeader("upload-from", FCImport.options.serviceType);  //No I18N
                          xhr.setRequestHeader("x-filename", encodeURIComponent(filedetails.newName + filedetails.fileExtn));    //No I18N
                          xhr.setRequestHeader("x-doc_conversion", true); //No I18N
                          xhr.setRequestHeader("x-content_type", filedetails.documentType);   //No I18N
                          xhr.setRequestHeader("x-uploadTo", 'myfolders');    //No I18N
                          xhr.setRequestHeader("x-folderId", "-1");   //No I18N
                          xhr.withCredentials = true;
                          xhr.send(filedetails.xfile[0].files[0]);
                      }
                  };
              })(jQuery);
